Syntax of terms

term ::= 
       | symbol
       | symbol(term, ..., term)

       | constant
       | - term
       | term + term
       | term - term
       | term * term

       | term[term]
       | term[term := term]

       | if prop then term else term fi

prop ::= 
       | term = term
       | term <> term
       | term < term
       | term <= term

       | true
       | false
       | not prop
       | prop and prop
       | prop or prop
       | prop => prop
       | if prop then prop else prop fi
